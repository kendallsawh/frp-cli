SELECT individuals.id as 'ref_no', applications.id as 'app_ref', addresses.id as 'addr_ref', individuals.f_name, individuals.m_name,individuals.l_name, individuals.dob, applications.old_registration_num as 'farmer_reg_num', addresses.house_num, addresses.road, districts.district as 'district name' FROM individuals join application_individual on individuals.id = application_individual.ind_id join `applications` on application_individual.app_id = applications.id join parcels on applications.id = parcels.application_id join land on parcels.land_id = land.id join tenure_codes on parcels.tenure_code_id = tenure_codes.id join addresses on land.address_id = addresses.id join districts on addresses.district_id = districts.id join wards on districts.ward_id = wards.id join counties on wards.county_id = counties.id WHERE districts.id = 233 and tenure_codes.id not in (7,5,6,8)
